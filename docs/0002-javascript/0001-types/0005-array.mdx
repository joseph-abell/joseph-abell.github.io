# Array

Arrays are lists of values. The values can be anything at all, and they don't need to match types. 

## Create

```JS
const a = ['a', 0, () => ({})] // ['a', 0, () => ({})]
const b = new Array('a', 'b', 'c') // ['a', 'b', 'c']
```

## Read

```JS
const a = ['a', 'b', 'c']
a[0]; // 'a'

a; // ['a', 'b', 'c'];
a[arr.length - 1] // 'c';
```

## Check how many items are in the array

```JS
const a = ['a', 'b', 'c'];
a.length; // 3 - note it isn't a function it's just a parameter on arrays
```

## Add an item to an array

```JS
const a = ['a', 'b', 'c'];
const b = [...a, b];

a; // ['a', 'b', 'c'];
b; // ['a', 'b', 'c', 'd'];
```

or:

```JS
const a = ['a', 'b', 'c', 'd'];
const b = a.concat(['e']);

a; // ['a', 'b', 'c', 'd']
b; // ['a', 'b', 'c', 'd', 'e']
```

## Filter on the values in the array

```JS
const ages = [12, 23, 34, 45, 56]

const adults = ages.filter(a => a >= 18);

ages; // [12, 23, 34, 45, 56]
adults; // [23, 34, 45, 56]
```

## Make a change to every value in an array

```JS
const a = [12, 23, 34, 45, 56]

const b = a.map(i => i + 1);

a; // [12, 23, 34, 45, 56]
b; // [13, 24, 35, 46, 57]
```

## Make a change to some values in the array if they meet a condition

```JS
const a = [12, 23, 34, 45, 56];

const lieAboutAge = a => a > 40 ? 40 : a;

const b = a.map(lieAboutAge);

a; // [12, 23, 34, 45, 56];
b; // [12, 23, 34, 40, 40];
```

## Flatten multiple arrays into one

```JS
const a = [[1, 2, 3, 4], 5];

const b = a.flat();

a; // [[1, 2, 3, 4], 5]
b; // [1, 2, 3, 4, 5]
```
